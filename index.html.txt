// ── Helpers ──────────────────────────────────────────────────
function catClass(cat) {
  if (cat === 'Wire Rope') return 'wr';
  if (cat === 'Green Pin') return 'gp';
  return 'wb';
}
 
function stockClass(qty) {
  if (qty <= 0) return 'empty';
  if (qty <= 5) return 'low';
  return 'ok';
}
 
function stockLabel(qty, unit) {
  if (qty <= 0) return `0 ${unit} ⚠`;
  return `${qty} ${unit}`;
}
 
// ── Render items list ─────────────────────────────────────────
function renderList(data) {
  const el = document.getElementById('items-list');
  const info = document.getElementById('results-info');
  info.textContent = `Showing ${data.length} of ${items.length} items`;
  if (!data.length) {
    el.innerHTML = `<div class="empty"><div class="icon">🔍</div><p>No items match your search.</p></div>`;
    return;
  }
  el.innerHTML = data.map(item => `
    <div class="item-card ${catClass(item.category)}" onclick="openModal('${item.sku}')">
      <div class="card-top">
        <span class="sku-badge">${item.sku}</span>
        <span class="stock-pill ${stockClass(item.stock)}">${stockLabel(item.stock, item.unit)}</span>
      </div>
      <div class="card-desc">${item.description.substring(0,100)}${item.description.length>100?'…':''}</div>
      <div class="card-meta">
        <span class="cat-tag ${catClass(item.category)}">${item.category}</span>
        ${item.code ? `<span>Code: ${item.code}</span>` : ''}
      </div>
    </div>
  `).join('');
}
 
// ── Search ────────────────────────────────────────────────────
function doSearch() {
  const q = document.getElementById('search-input').value.toLowerCase();
  const btn = document.getElementById('clear-btn');
  btn.classList.toggle('visible', q.length > 0);
 
  let base = currentCat === 'all' ? items : items.filter(i => i.category === currentCat);
  filtered = q ? base.filter(i =>
    i.sku.toLowerCase().includes(q) ||
    i.description.toLowerCase().includes(q) ||
    i.code.toLowerCase().includes(q)
  ) : base;
  renderList(filtered);
}
 
function clearSearch() {
  document.getElementById('search-input').value = '';
  document.getElementById('clear-btn').classList.remove('visible');
  doSearch();
}
 
function filterCat(cat, el) {
  currentCat = cat;
  document.querySelectorAll('.chip').forEach(c => {
    c.classList.add('inactive');
    c.classList.remove('all','wr','gp','wb');
  });
  el.classList.remove('inactive');
  if (cat === 'all') el.classList.add('all');
  else if (cat === 'Wire Rope') el.classList.add('wr');
  else if (cat === 'Green Pin') el.classList.add('gp');
  else el.classList.add('wb');
  doSearch();
}
 
// ── Modal ─────────────────────────────────────────────────────
let currentQR = null;
 
function openModal(sku) {
  const item = items.find(i => i.sku === sku);
  if (!item) return;
 
  document.getElementById('m-sku').textContent = item.sku;
  document.getElementById('m-desc').textContent = item.description;
  document.getElementById('m-stock').textContent = `${item.stock} ${item.unit}`;
  document.getElementById('m-opening').textContent = `${item.opening} ${item.unit}`;
  document.getElementById('m-unit').textContent = item.unit;
  document.getElementById('m-cat').textContent = item.category;
 
  // Generate QR
  const qrDiv = document.getElementById('qr-container');
  qrDiv.innerHTML = '';
  new QRCode(qrDiv, {
    text: item.sku,
    width: 160, height: 160,
    colorDark: '#1F4E79',
    colorLight: '#ffffff',
    correctLevel: QRCode.CorrectLevel.M
  });
 
  document.getElementById('modal-overlay').classList.add('open');
}
 
function closeModal(e) {
  if (e.target === document.getElementById('modal-overlay')) closeModalDirect();
}
function closeModalDirect() {
  document.getElementById('modal-overlay').classList.remove('open');
}
 
// ── Labels page ───────────────────────────────────────────────
let labelsGenerated = false;
 
function genLabels(cat) {
  const data = cat === 'all' ? items : items.filter(i => i.category === cat);
  const grid = document.getElementById('labels-grid');
  grid.innerHTML = '<div style="grid-column:1/-1;text-align:center;padding:20px;color:#888;font-size:13px;">⏳ Generating QR codes...</div>';
 
  setTimeout(() => {
    grid.innerHTML = '';
    data.forEach((item, idx) => {
      const card = document.createElement('div');
      card.className = 'label-card';
      const tagColor = item.category === 'Wire Rope' ? '#2E75B6' :
                       item.category === 'Green Pin' ? '#375623' : '#7030A0';
      const tagBg = item.category === 'Wire Rope' ? '#BDD7EE' :
                    item.category === 'Green Pin' ? '#E2EFDA' : '#EAD7F7';
 
      card.innerHTML = `
        <div class="label-cat" style="background:${tagBg};color:${tagColor}">${item.category}</div>
        <div class="label-qr" id="lqr-${idx}"></div>
        <div class="label-sku">${item.sku}</div>
        <div class="label-desc-short">${item.description.substring(0,60)}${item.description.length>60?'…':''}</div>
      `;
      grid.appendChild(card);
 
      setTimeout(() => {
        new QRCode(document.getElementById(`lqr-${idx}`), {
          text: item.sku,
          width: 80, height: 80,
          colorDark: tagColor,
          colorLight: '#ffffff',
          correctLevel: QRCode.CorrectLevel.L
        });
      }, idx * 5);
    });
  }, 50);
}
 
// ── Upload handler ────────────────────────────────────────────
function handleUpload(input) {
  const file = input.files[0];
  if (!file) return;
  // Show success (in real implementation, would parse XLSX)
  const status = document.getElementById('upload-status');
  status.style.display = 'block';
  status.textContent = `✅ "${file.name}" loaded! In the full version, items would refresh from this file.`;
  setTimeout(() => status.style.display = 'none', 4000);
}
 
// ── Pages ─────────────────────────────────────────────────────
function showPage(id, tab) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  document.getElementById('page-' + id).classList.add('active');
  tab.classList.add('active');
 
  if (id === 'labels' && !labelsGenerated) {
    genLabels('all');
    labelsGenerated = true;
  }
}
